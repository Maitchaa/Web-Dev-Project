-- --------------------------------------------------------
--
-- Table structure for table `plant`
--

CREATE TABLE IF NOT EXISTS plant (
    plant_id INT NOT NULL,
    scientific_name varchar(50) NOT NULL,
    common_name varchar(50) NOT NULL,
    intro TEXT NOT NULL,
    watering TEXT NOT NULL,
    watering_frequency TEXT NOT NULL,
    watering_technique TEXT NOT NULL,
    soil TEXT NOT NULL,
    light TEXT NOT NULL,
    temperature TEXT NOT NULL,
    fertilization TEXT NOT NULL,
    pruning TEXT NOT NULL,
    humidity TEXT NOT NULL,
    air_circulation TEXT NOT NULL,
    pest_management TEXT NOT NULL,
    repotting TEXT NOT NULL,
    image_path varchar(15) NOT NULL, 
    sticker_path varchar(15) NOT NULL, 
    PRIMARY KEY(plant_id)
);


-- --------------------------------------------------------
--
-- Table structure for table `resources`
--

CREATE TABLE IF NOT EXISTS resources (
    plant_id INT NOT NULL,
    resource_name varchar(150) NOT NULL,
    resource_href varchar(255) NOT NULL,
    PRIMARY KEY(plant_id, resource_name),
    FOREIGN KEY(plant_id) REFERENCES plant(plant_id)
);


-- --------------------------------------------------------
--
-- Table structure for table `account`
--
CREATE TABLE IF NOT EXISTS account (
    account_id INT NOT NULL,
    username varchar(30) NOT NULL UNIQUE,
    email varchar(70) NOT NULL UNIQUE,
    password_txt varchar(20) NOT NULL CHECK(CHAR_LENGTH(password_txt) >= 8),
    profile_pc varchar(30) DEFAULT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY(account_id)
);

-- --------------------------------------------------------
--
-- Table structure for table `collections`
--

CREATE TABLE IF NOT EXISTS collections (
    account_id INT NOT NULL,
    plant_id INT NOT NULL,
    PRIMARY KEY(account_id, plant_id),
    FOREIGN KEY(account_id) REFERENCES account(account_id),
    FOREIGN KEY(plant_id) REFERENCES plant(plant_id)
);

-- --------------------------------------------------------

--
-- Table structure for table `reminder`
--

CREATE TABLE IF NOT EXISTS reminder (
    reminder_id INT NOT NULL,
    reminder_title varchar(100) NOT NULL,
    reminder_note TEXT NOT NULL,
    reminder_time DATETIME NOT NULL,
    frequency VARCHAR(20),
    notification_type ENUM('daily', 'weekly', 'monthly'),
    is_active BOOLEAN NOT NULL,
    account_id int(3) NOT NULL,
    plant_id int(3) NOT NULL,
    PRIMARY KEY(reminder_id),
    FOREIGN KEY(account_id) REFERENCES account(account_id),
    FOREIGN KEY(plant_id) REFERENCES plant(plant_id)
);
